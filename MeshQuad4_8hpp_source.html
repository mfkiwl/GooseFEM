678 inline xt::xtensor<size_t, 1> FineLayer::elementgrid_ravel(
679  std::vector<size_t> start_stop_rows,
680  std::vector<size_t> start_stop_cols) const
681 {
682  GOOSEFEM_ASSERT(start_stop_rows.size() == 0 || start_stop_rows.size() == 2);
683  GOOSEFEM_ASSERT(start_stop_cols.size() == 0 || start_stop_cols.size() == 2);
684 
685  std::array<size_t, 2> rows;
686  std::array<size_t, 2> cols;
687 
688  if (start_stop_rows.size() == 2) {
689  std::copy(start_stop_rows.begin(), start_stop_rows.end(), rows.begin());
690  GOOSEFEM_ASSERT(rows[0] <= this->nely());
691  GOOSEFEM_ASSERT(rows[1] <= this->nely());
692  }
693  else {
694  rows[0] = 0;
695  rows[1] = this->nely();
696  }
697 
698  if (start_stop_cols.size() == 2) {
699  std::copy(start_stop_cols.begin(), start_stop_cols.end(), cols.begin());
700  GOOSEFEM_ASSERT(cols[0] <= this->nelx());
701  GOOSEFEM_ASSERT(cols[1] <= this->nelx());
702  }
703  else {
704  cols[0] = 0;
705  cols[1] = this->nelx();
706  }
707 
708  if (rows[0] == rows[1] || cols[0] == cols[1]) {
709  xt::xtensor<size_t, 1> ret = xt::empty<size_t>({0});
710  return ret;
711  }
712 
713  // Compute dimensions
714 
715  auto H = xt::cumsum(m_nhy);
716  size_t yl = 0;
717  if (rows[0] > 0) {
718  yl = xt::argmax(H > rows[0])();
719  }
720  size_t yu = xt::argmax(H >= rows[1])();
721  size_t hx = std::max(m_nhx(yl), m_nhx(yu));
722  size_t xl = (cols[0] - cols[0] % hx) / hx;
723  size_t xu = (cols[1] - cols[1] % hx) / hx;
724 
725  // Allocate output
726 
727  size_t N = 0;
728 
729  for (size_t i = yl; i <= yu; ++i) {
730  // no refinement
731  size_t n = (xu - xl) * hx / m_nhx(i);
732  // refinement
733  if (m_refine(i) != -1) {
734  n *= 4;
735  }
736  N += n;
737  }
738 
739  xt::xtensor<size_t, 1> ret = xt::empty<size_t>({N});
740 
741  // Write output
742 
743  N = 0;
744 
745  for (size_t i = yl; i <= yu; ++i) {
746  // no refinement
747  size_t n = (xu - xl) * hx / m_nhx(i);
748  size_t h = hx;
749  // refinement
750  if (m_refine(i) != -1) {
751  n *= 4;
752  h *= 4;
753  }
754  xt::xtensor<size_t, 1> e = m_startElem(i) + xl * h / m_nhx(i) + xt::arange<size_t>(n);
755  xt::view(ret, xt::range(N, N + n)) = e;
756  N += n;
757  }
758 
759  return ret;
760 }

1126  eleft--;

1150
