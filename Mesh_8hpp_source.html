52 inline xt::xtensor<size_t, 2> overlapping(
53  const xt::xtensor<double, 2>& coor_a,
54  const xt::xtensor<double, 2>& coor_b,
55  double rtol,
56  double atol)
57 {
58  GOOSEFEM_ASSERT(coor_a.shape(1) == coor_b.shape(1));
59 
60  std::vector<size_t> ret_a;
61  std::vector<size_t> ret_b;
62

64

66  coor_b, xt::view(coor_a, i, xt::all()), rtol, atol), 1)));
67 
68  for (auto& j : idx) {

72  }

75  for (size_t i = 0; i < ret_a.size(); ++i) {
76  ret(0, i) = ret_a[i];
77  ret(1, i) = ret_b[i];
78  }
