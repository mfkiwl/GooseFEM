343 inline void MatrixPartitionedTyingsSolver<Solver>::solve(
344  MatrixPartitionedTyings& matrix, const xt::xtensor<double, 2>& b, xt::xtensor<double, 2>& x)
345 {
346  GOOSEFEM_ASSERT(xt::has_shape(b, {matrix.m_nnode, matrix.m_ndim}));
347  GOOSEFEM_ASSERT(xt::has_shape(x, {matrix.m_nnode, matrix.m_ndim}));
348 
349  this->factorize(matrix);
350 
351  Eigen::VectorXd B_u = matrix.AsDofs_u(b);
352  Eigen::VectorXd B_d = matrix.AsDofs_d(b);
353  Eigen::VectorXd X_p = matrix.AsDofs_p(x);
354 
355  B_u += matrix.m_Cud * B_d;
356 
357  Eigen::VectorXd X_u = m_solver.solve(Eigen::VectorXd(B_u - matrix.m_ACup * X_p));
358  Eigen::VectorXd X_d = matrix.m_Cdu * X_u + matrix.m_Cdp * X_p;
359 
360  #pragma omp parallel for
361  for (size_t m = 0; m < matrix.m_nnode; ++m) {
362  for (size_t i = 0; i < matrix.m_ndim; ++i) {
363  if (matrix.m_dofs(m, i) < matrix.m_nnu) {
364  x(m, i) = X_u(matrix.m_dofs(m, i));
365  }
366  else if (matrix.m_dofs(m, i) >= matrix.m_nni) {
367  x(m, i) = X_d(matrix.m_dofs(m, i) - matrix.m_nni);
368  }
369  }
370  }
