224 inline void Quadrature::compute_dN()
225 {
226  #pragma omp parallel
227  {
228  xt::xtensor<double, 2> J = xt::empty<double>({2, 2});
229  xt::xtensor<double, 2> Jinv = xt::empty<double>({2, 2});
230 
231  #pragma omp for
232  for (size_t e = 0; e < m_nelem; ++e) {
233 
234  auto x = xt::adapt(&m_x(e, 0, 0), xt::xshape<m_nne, m_ndim>());
235 
236  for (size_t q = 0; q < m_nip; ++q) {
237 
238  auto dNxi = xt::adapt(&m_dNxi(q, 0, 0), xt::xshape<m_nne, m_ndim>());
239  auto dNx = xt::adapt(&m_dNx(e, q, 0, 0), xt::xshape<m_nne, m_ndim>());
240 
241  // J(i,j) += dNxi(m,i) * x(m,j);
242  J(0, 0) = dNxi(0, 0) * x(0, 0) + dNxi(1, 0) * x(1, 0) + dNxi(2, 0) * x(2, 0) +
243  dNxi(3, 0) * x(3, 0);
244  J(0, 1) = dNxi(0, 0) * x(0, 1) + dNxi(1, 0) * x(1, 1) + dNxi(2, 0) * x(2, 1) +
245  dNxi(3, 0) * x(3, 1);
246  J(1, 0) = dNxi(0, 1) * x(0, 0) + dNxi(1, 1) * x(1, 0) + dNxi(2, 1) * x(2, 0) +
247  dNxi(3, 1) * x(3, 0);
248  J(1, 1) = dNxi(0, 1) * x(0, 1) + dNxi(1, 1) * x(1, 1) + dNxi(2, 1) * x(2, 1) +
249  dNxi(3, 1) * x(3, 1);
250 
251  double Jdet = detail::inv(J, Jinv);
252 
253  // dNx(m,i) += Jinv(i,j) * dNxi(m,j);
254  for (size_t m = 0; m < m_nne; ++m) {
255  dNx(m, 0) = Jinv(0, 0) * dNxi(m, 0) + Jinv(0, 1) * dNxi(m, 1);
256  dNx(m, 1) = Jinv(1, 0) * dNxi(m, 0) + Jinv(1, 1) * dNxi(m, 1);
257  }
258 
259  m_vol(e, q) = m_w(q) * Jdet;
260  }
261  }
262  }
